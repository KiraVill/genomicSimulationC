411static inline int has_same_alleles(char* p1, char* p2, int i) {
412 return (p1[i<<1] == p2[i<<1] || p1[(i<<1) + 1] == p2[i] || p1[i] == p2[(i<<1) + 1]);
413}
414// w is window length, i is start value
428static inline int has_same_alleles_window(char* g1, char* g2, int start, int w) {
429 int same = TRUE;
430 int i;
431 for (int j = 0; j < w; ++j) {
432 i = start + j;
433 same = same && (g1[i<<1] == g2[i<<1] || g1[(i<<1) + 1] == g2[i] || g1[i] == g2[(i<<1) + 1]);
434 }
435 return same;
436}
