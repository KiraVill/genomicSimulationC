703static inline int has_same_alleles(const char* p1, const char* p2, const int i) {
704 return (p1[i<<1] == p2[i<<1] || p1[(i<<1) + 1] == p2[i] || p1[i] == p2[(i<<1) + 1]);
705}
706// w is window length, i is start value
720static inline int has_same_alleles_window(const char* g1, const char* g2, const int start, const int w) {
721 int same = TRUE;
722 int i;
723 for (int j = 0; j < w; ++j) {
724 i = start + j;
725 same = same && (g1[i<<1] == g2[i<<1] || g1[(i<<1) + 1] == g2[i] || g1[i] == g2[(i<<1) + 1]);
726 }
727 return same;
728}
