461{ size_t len = nentries > n##cap ? n##cap : nentries; as = gsc_malloc_wrap(sizeof(n##stack[0])*len,GSC_TRUE); memcpy(as,n,sizeof(n##stack[0])*len); } } while (0)

495#define GSC_STRETCH_BUFFER(n,newlen) do { \
496 if (newlen < n##cap) { } \
497 else if (n##cap >= sizeof(n##stack)/sizeof(n##stack[0])) { \
